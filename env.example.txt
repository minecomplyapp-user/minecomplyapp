# MineComply Mobile App - Environment Variables Configuration
# Copy this file to .env and fill in your actual values

# =============================================================================
# SUPABASE CONFIGURATION (REQUIRED)
# =============================================================================
# Get these from your Supabase project dashboard:
# https://app.supabase.com/project/YOUR_PROJECT/settings/api

# Supabase Project URL
EXPO_PUBLIC_SUPABASE_URL=https://your-project.supabase.co

# Supabase Anon/Public Key
EXPO_PUBLIC_SUPABASE_ANON_KEY=eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...

# =============================================================================
# API CONFIGURATION (OPTIONAL)
# =============================================================================
# The app will automatically determine the API URL based on the build type:
# - Development (Expo Go): Uses local API or auto-detected dev server
# - Production (APK): Uses production API or fallback

# Production API URL (for APK builds)
# If not set, defaults to: https://minecomplyapi.onrender.com/api
PRODUCTION_API_BASE_URL=https://minecomplyapi.onrender.com/api

# Local API URL (for development)
# If not set, auto-detects from Expo dev server host on port 3000
LOCAL_API_BASE_URL=http://192.168.1.100:3000/api

# =============================================================================
# EMAIL CONFIRMATION REDIRECT (OPTIONAL)
# =============================================================================
# Where to redirect users after email confirmation
# Default: https://kwiruu.github.io/minecomplyconfirmpage/
EXPO_PUBLIC_CONFIRMATION_REDIRECT_URL=https://kwiruu.github.io/minecomplyconfirmpage/

# =============================================================================
# NOTES
# =============================================================================
# - EXPO_PUBLIC_* variables are accessible in the app code
# - Other variables are only available at build time
# - For APK builds, these values are baked in at build time
# - Changes require rebuilding the APK
# - Never commit .env to version control (it's in .gitignore)
